> 
> e.g. something like the following (html form POST's to this script):
> 
> import os
> import cgi
> from mod_python import apache
> 
> def handler(req):
> 	req.content_type  = "text/plain"
> 	req.send_http_header()
> 	
> 	req.add_common_vars()
