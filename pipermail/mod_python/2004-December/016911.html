Greetings!

Try this:

    <Directory "/Users/amarre/Sites/web-app/">
        AddHandler mod_python .py .psp
        PythonDebug On
        <Files "*.py">
            PythonHandler hello
        </Files>
        <Files "*.psp">
            PythonHandler mod_python.psp
        </Files>
    </Directory>
