  I came up with this:
  
 
  
from mod_python import apache
  
def handler(req):
    req.content_type = 'text/plain'
    req.add_common_vars()
    req.write('IP address of requester is ' + req.subprocess_env['REMOTE_ADDR']);
    return apache.OK
