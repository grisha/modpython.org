def GetSid(req, sid=""):
  sess = Session(req);
  if sess.is_new():
    sess['u']="user";
    sess['p']="password";
    sid = sess.id();
    sess.save;
  else:
    sid = sess.id();
  req.write("%s" % sid);
  return;

def UseSession(req, sid=""):
  req.write("%s" % sid);
  sess = Session(req, sid, None, 1800, 0);
  if sess.is_new():
