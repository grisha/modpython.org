def authenhandler( req ) :
    req.session = Session.Session(req, timeout=300)
    if req.session.is_new() :
        if 'eggs' == req.user and 'spam' == req.get_basic_auth_pw() :
            req.session.save()
        else:
            return apache.HTTP_UNAUTHORIZED
    return apache.OK
