    PythonDebug On
    SetHandler mod_python

    PythonHeaderParserHandler Example::handler
    PythonPath "sys.path + ['/home/dgc/lib/python']"
</Location>

unix$ cat Example.py
from mod_python import apache

        return apache.DECLINED

    print
    cgi.print_environ()

if not os.isatty(0):
    cgitb.enable()
if __name__ == "__main__":
    main()
