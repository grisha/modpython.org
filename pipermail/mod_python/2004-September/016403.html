from mod_python import apache, psp

def handler(req):
    req.content_type = "text/html"
    req.write("hi<br>")
    req.write(req.filename)
    template = psp.PSP(req, filename='template.html')
    template.run({'what':'world'})
    req.write('<br>after')
    return apache.OK
