If i understand good what you are trying to do the code will be :

from mod_python import apache

def handler(req):
	objInst = MyObject(req)
	req.write(str(objInst))
	return apache.OK

class MyObject:
	def __init__(self, req):
		self.req = req

	def __str__(self):
		return "some string"
