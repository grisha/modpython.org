import sys

dict1 = {
            '0' : 'zero',
            '1' : 'one',
            '2' : 'two',
            '3' : 'three',
            '4' : 'four',
            '5' : 'five',
            '6' : 'six',
            '7' : 'seven',
            '8' : 'eight',
            '9' : 'nine',
            ' ' : '\n',
        }

def convert(numbers):
    w = []
    for argument in numbers:
        w.append([dict1[digit] for digit in argument])
