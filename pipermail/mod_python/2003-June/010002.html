def getTime(host):
     t = "Error"
     try:
         s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
         s.connect((host,8000))
         s.send('DAMELAAHORA\0')
         t = receive(s,15)
     finally:
         if s != None:
             s.close()

def receive(socket, size):
     buffer=''
     while size != 0:
         data = socket.recv(size)
         buffer += data
         size = size-len(data)
     return buffer
