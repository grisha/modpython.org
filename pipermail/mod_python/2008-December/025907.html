from os.path import basename

   req.write("""
      req.filename: %s
      req.uri: %s
      req.path_info: %s
      req.args: %s
      """ % (req.filename, req.uri, req.path_info, req.args)
   )
   if basename(req.filename) == 'index.py':
      req.uri = '/'
   req.write('basename: "%s" req.uri: "%s"' % (basename(req.filename), req.uri))

   return publisher.handler(req)
